function solution(numbers, target) {
  let cnt = 0;
  const dfs = (depth, curr, value) => {
    if (depth === numbers.length) {
      if (value === 0) cnt++;
      return;
    }

    dfs(depth + 1, curr + 1, value - numbers[curr]);
    dfs(depth + 1, curr + 1, value + numbers[curr]);
  };

  dfs(0, 0, target);

  return cnt;
}
